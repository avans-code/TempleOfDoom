The Factory Pattern

Creational Pattern
LevelFactory.cs 

public class LevelFactory {
    public Level CreateLevel(LevelDto dto) {
        Level level = new Level();
        foreach (var roomDto in dto.Rooms) {
            Room room = new Room(roomDto.Id, roomDto.Width, roomDto.Height);
            // Polymorfisme: Factory kiest de juiste implementatie
            ParseEnemies(room, roomDto.Enemies);
            level.Rooms.Add(room.Id, room);
        }
        return level;
    }
}

The Observer Pattern

Ontkoppelde communicatie via Events

Behavioral Pattern

// PressurePlate: De Publisher
public event Action? OnOccupancyChanged;
public bool IsOccupied {
    set { 
        if (_isOccupied != value) {
            _isOccupied = value; OnOccupancyChanged?.Invoke(); 
        }
    }
}
// GameController: De Subscriber
plate.OnOccupancyChanged += () => {
    if (plates.All(p => p.IsOccupied)) UnlockDoors();
};

The Decorator Pattern

Gedrag stapelen zonder inheritance hell.

Structural Pattern

public class OpenOnOddDoor : IDoor {
    private readonly IDoor _innerDoor; // Wrapped object

    public bool CanEnter(Player player, Room room) {
        // Decorator logica
        if (player.Lives % 2 == 0) return false;
        
        // Delegeer naar de basis-deur
        return _innerDoor.CanEnter(player, room);
    }
}

The Adapter Pattern

Koppelt de onstabiele External.dll aan onze Entity base class.


public class EnemyAdapter : Entity {
    private readonly ExternalEnemy _dllEnemy;

    public override void Move() {
        _dllEnemy.UpdatePosition(); // Vertaling
        this.X = _dllEnemy.CurrentX;
    }
}
