The Factory Pattern

Creational
LevelFactory.cs 

public Level CreateLevel(LevelDto dto)
{
    var level = new Level { Player = new Player(dto.Player.StartX, dto.Player.StartY, dto.Player.Lives) };

    foreach (var roomDto in dto.Rooms)
    {
        var room = new Room(roomDto.Id, roomDto.Width, roomDto.Height);
        
        // Factory verwerkt special tiles zoals binnenmuren
        if (roomDto.SpecialFloorTiles != null)
            foreach (var tile in roomDto.SpecialFloorTiles)
                room.SpecialTiles.Add((tile.X, tile.Y), tile.Type);

        // Koppeling met Adapter voor DLL Enemies
        ParseEnemies(room, roomDto.Enemies);
        level.Rooms.Add(room.Id, room);
    }
    return level;
}

The Observer Pattern

Behavioral
PressurePlate.cs | GameController.cs 

// Subject: PressurePlate vuurt event bij verandering
public event Action? OnOccupancyChanged;
public bool IsOccupied
{
    set { 
        if (_isOccupied == value) return;
        _isOccupied = value;
        OnOccupancyChanged?.Invoke(); 
    }
}

// Observer: GameController luistert en reageert
private void WireObservers()
{
    foreach (var plate in currentRoom.Plates)
    {
        plate.OnOccupancyChanged += () => {
            if (plates.All(p => p.IsOccupied)) UnlockSwitchDoors();
        };
    }
}

The Decorator Pattern

Structural
OpenOnOddDoor.cs 

public class OpenOnOddDoor : IDoor
{
    private readonly IDoor _innerDoor; // De 'wrapped' component
    private readonly Level _level;

    public bool CanEnter(Player player, Room room)
    {
        // 1. Eigen logica van de Decorator
        if (player.Lives % 2 == 0) return false;

        // 2. Delegeer naar de basis-deur (bijv. ToggleDoor)
        return _innerDoor.CanEnter(player, room);
    }

    public void Unlock() => _innerDoor.Unlock();
}

The Adapter Pattern

Structural
EnemyAdapter.cs 

public class EnemyAdapter : Entity
{
    private readonly CODE_TempleOfDoom_DownloadableContent.Enemy _dllEnemy;

    public override void Move()
    {
        // Vertaal actie naar de onstabiele school-DLL
        _dllEnemy.Move();
        
        // Synchroniseer coördinaten met ons eigen systeem
        this.X = _dllEnemy.X;
        this.Y = _dllEnemy.Y;
    }
}
