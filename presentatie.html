Factory Pattern (10p)

Level Factory

LevelFactory.cs

public Level CreateLevel(LevelDto dto)
{
    var level = new Level();
    foreach (var roomDto in dto.Rooms)
    {
        // Polymorfisme: Room bepaalt eigen interne logica
        var room = new Room(roomDto.Id, roomDto.Width, roomDto.Height);
        
        // Factory ontkoppelt de DTO van het rijke Domain Model
        ParseEnemies(room, roomDto.Enemies);
        level.Rooms.Add(room.Id, room);
    }
    return level;
}
Observer Pattern (10p)

Event-Driven Observer

PressurePlate.cs

// Het Subject: Meldt veranderingen aan abonnees
public event Action? OnOccupancyChanged;

public bool IsOccupied 
{
    set {
        if (_isOccupied == value) return;
        _isOccupied = value;
        // Notificeer de GameController (Observer)
        OnOccupancyChanged?.Invoke();
    }
}
Decorator Pattern (10p)

Door Decorator

OpenOnOddDoor.cs

public class OpenOnOddDoor : IDoor
{
    private readonly IDoor _innerDoor; // Ingepakte component

    public bool CanEnter(Player player, Room room)
    {
        // 1. Extra gedrag: Check op levens
        if (player.Lives % 2 == 0) return false;

        // 2. Delegeer naar de oorspronkelijke deur
        return _innerDoor.CanEnter(player, room);
    }
}
Adapter Pattern (10p)

DLL Adapter

EnemyAdapter.cs

public class EnemyAdapter : Entity
{
    private readonly ExternalDLL.Enemy _dllEnemy;

    public override void Move()
    {
        // Vertaal acties naar de onstabiele school-DLL
        _dllEnemy.Move();
        
        // Sync coördinaten tussen systemen
        this.X = _dllEnemy.X;
        this.Y = _dllEnemy.Y;
    }
}
